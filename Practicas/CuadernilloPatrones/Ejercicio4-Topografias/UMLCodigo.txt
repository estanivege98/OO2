@startuml
interface Topografia <<Component>> {
  + proporcionAgua(): Real
  + proporcionTierra(): Real
  + esIgualA(t: Topografia): Bool
  + esIgualAAgua(t: TopografiaAgua): Bool
  + esIgualATierra(t: TopografiaTierra): Bool
  + esIgualAPantano(t: TopografiaPantano): Bool
  + esIgualAMixta(t: TopografiaMixta): Bool
}

class TopografiaAgua <<Leaf>> {
  + proporcionAgua(): Real
  + proporcionTierra(): Real
  + esIgualA(t: Topografia): Bool
  + esIgualAAgua(t: TopografiaAgua): Bool
}

class TopografiaTierra <<Leaf>> {
  + proporcionAgua(): Real
  + proporcionTierra(): Real
  + esIgualA(t: Topografia): Bool
  + esIgualATierra(t: TopografiaTierra): Bool
}

class TopografiaPantano <<Leaf>> {
  + proporcionAgua(): Real
  + proporcionTierra(): Real
  + esIgualA(t: Topografia): Bool
  + esIgualAPantano(t: TopografiaPantano): Bool
}

class TopografiaMixta <<Composite>> {
  - componentes: List<Topografia>
  + <<create>>TopografiaMixta()
  + agregarComponente(t: Topografia)
  + proporcionAgua(): Real
  + proporcionTierra(): Real
  + esIgualA(t: Topografia): Bool
  + esIgualAMixta(t: TopografiaMixta): Bool
}

TopografiaAgua ..|> Topografia
TopografiaTierra ..|> Topografia
TopografiaPantano ..|> Topografia
TopografiaMixta ..|> Topografia
TopografiaMixta "0..4" *-- Topografia : componentes
@enduml
