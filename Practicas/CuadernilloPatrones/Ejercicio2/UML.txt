#<<abstract>>calcularBasico(): Real
#<<abstract>>calcularAdicional(): Real
-calculoDescuento(): Real
}

-casado: Bool
-cantHijos: Integer

#calcularBasico(): Real
#calcularAdicional(): Real
}

-cantHijos: Integer
